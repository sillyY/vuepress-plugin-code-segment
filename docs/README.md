<template>
  <section>
    <div class="box-vue">{{num}}</div>
    <button @click="add">+</button>
    <button @click="reduce">-</button>
  </section>
</template>
<script>
  export default {
    data: () => ({ num: 1 }),
    methods: {
      add() { this.num += 1 },
      reduce() {
        this.num -= 1
      }
    }
  }
</script>
<style>
  .box-vue {
    color: red;
  }
</style>
